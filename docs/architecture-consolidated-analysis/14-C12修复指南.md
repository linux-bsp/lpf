# C-12 修复实施指南

## 目标

修复HAL层错误信息丢失问题，建立细粒度的错误码体系和错误上下文传递机制。

---

## 问题描述

### 当前问题

1. **错误码过于粗粒度**
   - 大多数错误都返回`OSAL_ERR_GENERIC`
   - 无法区分具体原因（设备忙、权限不足、硬件故障等）

2. **errno信息丢失**
   - 底层errno在LOG_ERROR中记录后丢失
   - 上层无法获取详细错误原因

3. **缺乏错误上下文**
   - 错误信息逐层丢失
   - 无法追踪错误来源

---

## 修复方案

### 1. 定义细粒度HAL错误码

**新增文件**: `core/hal/include/hal_error.h`

```c
#ifndef HAL_ERROR_H
#define HAL_ERROR_H

#include "osal_types.h"

/*===========================================================================
 * HAL错误码定义
 *===========================================================================*/

/* 基础错误码（-1000 ~ -1099） */
#define HAL_SUCCESS                 0
#define HAL_ERR_GENERIC            (-1000)  /* 通用错误 */
#define HAL_ERR_INVALID_PARAM      (-1001)  /* 无效参数 */
#define HAL_ERR_NOT_INITIALIZED    (-1002)  /* 未初始化 */
#define HAL_ERR_ALREADY_INITIALIZED (-1003) /* 已初始化 */
#define HAL_ERR_NO_MEMORY          (-1004)  /* 内存不足 */
#define HAL_ERR_TIMEOUT            (-1005)  /* 超时 */

/* 设备错误码（-1100 ~ -1199） */
#define HAL_ERR_DEVICE_NOT_FOUND   (-1100)  /* 设备未找到 */
#define HAL_ERR_DEVICE_BUSY        (-1101)  /* 设备忙 */
#define HAL_ERR_DEVICE_ERROR       (-1102)  /* 设备错误 */
#define HAL_ERR_DEVICE_OFFLINE     (-1103)  /* 设备离线 */

/* 权限错误码（-1200 ~ -1299） */
#define HAL_ERR_PERMISSION_DENIED  (-1200)  /* 权限不足 */
#define HAL_ERR_ACCESS_DENIED      (-1201)  /* 访问拒绝 */

/* I/O错误码（-1300 ~ -1399） */
#define HAL_ERR_IO_ERROR           (-1300)  /* I/O错误 */
#define HAL_ERR_READ_ERROR         (-1301)  /* 读取错误 */
#define HAL_ERR_WRITE_ERROR        (-1302)  /* 写入错误 */
#define HAL_ERR_OPEN_ERROR         (-1303)  /* 打开错误 */
#define HAL_ERR_CLOSE_ERROR        (-1304)  /* 关闭错误 */

/* 硬件错误码（-1400 ~ -1499） */
#define HAL_ERR_HARDWARE_FAULT     (-1400)  /* 硬件故障 */
#define HAL_ERR_BUS_ERROR          (-1401)  /* 总线错误 */
#define HAL_ERR_CHECKSUM_ERROR     (-1402)  /* 校验和错误 */
#define HAL_ERR_FRAMING_ERROR      (-1403)  /* 帧错误 */
#define HAL_ERR_OVERRUN            (-1404)  /* 溢出 */

/*===========================================================================
 * 错误上下文结构
 *===========================================================================*/

typedef struct {
    int32_t error_code;         /* HAL错误码 */
    int32_t system_errno;       /* 系统errno */
    const char *module;         /* 模块名 */
    const char *function;       /* 函数名 */
    const char *file;           /* 文件名 */
    int32_t line;               /* 行号 */
    uint32_t timestamp;         /* 时间戳 */
    char message[128];          /* 错误消息 */
} hal_error_context_t;

/*===========================================================================
 * 错误处理API
 *===========================================================================*/

/**
 * @brief 设置最后一次错误
 */
void HAL_SetLastError(int32_t error_code, int32_t system_errno,
                      const char *module, const char *function,
                      const char *file, int32_t line,
                      const char *format, ...);

/**
 * @brief 获取最后一次错误
 */
const hal_error_context_t* HAL_GetLastError(void);

/**
 * @brief 清除最后一次错误
 */
void HAL_ClearLastError(void);

/**
 * @brief 将errno映射到HAL错误码
 */
int32_t HAL_ErrnoToError(int32_t err);

/**
 * @brief 获取错误码的字符串描述
 */
const char* HAL_ErrorString(int32_t error_code);

/*===========================================================================
 * 辅助宏
 *===========================================================================*/

#define HAL_SET_ERROR(code, errno, fmt, ...) \
    HAL_SetLastError(code, errno, __MODULE__, __func__, __FILE__, __LINE__, fmt, ##__VA_ARGS__)

#define HAL_SET_ERROR_SIMPLE(code) \
    HAL_SetLastError(code, 0, __MODULE__, __func__, __FILE__, __LINE__, NULL)

#endif /* HAL_ERROR_H */
```

### 2. 实现错误处理模块

**新增文件**: `core/hal/src/hal_error.c`

```c
#include "hal_error.h"
#include "osal.h"
#include <errno.h>
#include <string.h>
#include <stdarg.h>
#include <stdio.h>

/*===========================================================================
 * 内部数据
 *===========================================================================*/

static __thread hal_error_context_t g_last_error = {0};

/*===========================================================================
 * errno映射表
 *===========================================================================*/

int32_t HAL_ErrnoToError(int32_t err)
{
    switch (err) {
        case 0:
            return HAL_SUCCESS;
        case ENOENT:
        case ENODEV:
            return HAL_ERR_DEVICE_NOT_FOUND;
        case EBUSY:
            return HAL_ERR_DEVICE_BUSY;
        case EACCES:
        case EPERM:
            return HAL_ERR_PERMISSION_DENIED;
        case ETIMEDOUT:
            return HAL_ERR_TIMEOUT;
        case ENOMEM:
            return HAL_ERR_NO_MEMORY;
        case EIO:
            return HAL_ERR_IO_ERROR;
        case EINVAL:
            return HAL_ERR_INVALID_PARAM;
        default:
            return HAL_ERR_GENERIC;
    }
}

/*===========================================================================
 * 错误字符串映射
 *===========================================================================*/

const char* HAL_ErrorString(int32_t error_code)
{
    switch (error_code) {
        case HAL_SUCCESS:
            return "Success";
        case HAL_ERR_GENERIC:
            return "Generic error";
        case HAL_ERR_INVALID_PARAM:
            return "Invalid parameter";
        case HAL_ERR_NOT_INITIALIZED:
            return "Not initialized";
        case HAL_ERR_ALREADY_INITIALIZED:
            return "Already initialized";
        case HAL_ERR_NO_MEMORY:
            return "Out of memory";
        case HAL_ERR_TIMEOUT:
            return "Timeout";
        case HAL_ERR_DEVICE_NOT_FOUND:
            return "Device not found";
        case HAL_ERR_DEVICE_BUSY:
            return "Device busy";
        case HAL_ERR_DEVICE_ERROR:
            return "Device error";
        case HAL_ERR_DEVICE_OFFLINE:
            return "Device offline";
        case HAL_ERR_PERMISSION_DENIED:
            return "Permission denied";
        case HAL_ERR_ACCESS_DENIED:
            return "Access denied";
        case HAL_ERR_IO_ERROR:
            return "I/O error";
        case HAL_ERR_READ_ERROR:
            return "Read error";
        case HAL_ERR_WRITE_ERROR:
            return "Write error";
        case HAL_ERR_OPEN_ERROR:
            return "Open error";
        case HAL_ERR_CLOSE_ERROR:
            return "Close error";
        case HAL_ERR_HARDWARE_FAULT:
            return "Hardware fault";
        case HAL_ERR_BUS_ERROR:
            return "Bus error";
        case HAL_ERR_CHECKSUM_ERROR:
            return "Checksum error";
        case HAL_ERR_FRAMING_ERROR:
            return "Framing error";
        case HAL_ERR_OVERRUN:
            return "Overrun";
        default:
            return "Unknown error";
    }
}

/*===========================================================================
 * 错误处理API实现
 *===========================================================================*/

void HAL_SetLastError(int32_t error_code, int32_t system_errno,
                      const char *module, const char *function,
                      const char *file, int32_t line,
                      const char *format, ...)
{
    g_last_error.error_code = error_code;
    g_last_error.system_errno = system_errno;
    g_last_error.module = module;
    g_last_error.function = function;
    g_last_error.file = file;
    g_last_error.line = line;
    g_last_error.timestamp = OSAL_GetTick();

    if (format != NULL) {
        va_list args;
        va_start(args, format);
        vsnprintf(g_last_error.message, sizeof(g_last_error.message), format, args);
        va_end(args);
    } else {
        snprintf(g_last_error.message, sizeof(g_last_error.message),
                 "%s", HAL_ErrorString(error_code));
    }
}

const hal_error_context_t* HAL_GetLastError(void)
{
    return &g_last_error;
}

void HAL_ClearLastError(void)
{
    memset(&g_last_error, 0, sizeof(g_last_error));
}
```

### 3. 修改HAL驱动使用新错误码

**示例：hal_serial_linux.c**

```c
// 之前
int32_t fd = OSAL_open(config->device, OSAL_O_RDWR, 0);
if (fd < 0) {
    LOG_ERROR("HAL_Serial", "Failed to open %s", config->device);
    return OSAL_ERR_GENERIC;
}

// 之后
int32_t fd = OSAL_open(config->device, OSAL_O_RDWR, 0);
if (fd < 0) {
    int32_t err = errno;
    int32_t hal_err = HAL_ErrnoToError(err);
    HAL_SET_ERROR(hal_err, err, "Failed to open %s: %s",
                  config->device, strerror(err));
    LOG_ERROR("HAL_Serial", "Failed to open %s: %s (errno=%d, hal_err=%d)",
              config->device, strerror(err), err, hal_err);
    return hal_err;
}
```

### 步骤3：添加单元测试（1天）

**新增文件**: `tests/unit/hal/test_hal_error.c`

```c
#include "hal_error.h"
#include <assert.h>
#include <errno.h>

void test_errno_mapping(void)
{
    assert(HAL_ErrnoToError(ENOENT) == HAL_ERR_DEVICE_NOT_FOUND);
    assert(HAL_ErrnoToError(EBUSY) == HAL_ERR_DEVICE_BUSY);
    assert(HAL_ErrnoToError(EACCES) == HAL_ERR_PERMISSION_DENIED);
    assert(HAL_ErrnoToError(ETIMEDOUT) == HAL_ERR_TIMEOUT);
}

void test_error_context(void)
{
    HAL_ClearLastError();
    
    HAL_SET_ERROR(HAL_ERR_DEVICE_NOT_FOUND, ENOENT,
                  "Device %s not found", "/dev/ttyS0");
    
    const hal_error_context_t *ctx = HAL_GetLastError();
    assert(ctx->error_code == HAL_ERR_DEVICE_NOT_FOUND);
    assert(ctx->system_errno == ENOENT);
    assert(strstr(ctx->message, "/dev/ttyS0") != NULL);
}

void test_error_string(void)
{
    assert(strcmp(HAL_ErrorString(HAL_SUCCESS), "Success") == 0);
    assert(strcmp(HAL_ErrorString(HAL_ERR_DEVICE_NOT_FOUND),
                  "Device not found") == 0);
}

int main(void)
{
    test_errno_mapping();
    test_error_context();
    test_error_string();
    return 0;
}
```
